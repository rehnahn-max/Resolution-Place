import { db } from "./firebase.js"; import { doc, getDoc } from "https://www.gstatic.com/firebasejs/10.12.5/firebase-firestore.js"; /* ========================= PEGAR ID DA URL ========================= */ const urlParams = new URLSearchParams(window.location.search); const id = urlParams.get("id"); const container = document.getElementById("produtoContainer"); async function carregarProduto() { if (!id) { container.innerHTML = "

Produto não encontrado

"; return; } try { const docRef = doc(db, "anuncios", id); const docSnap = await getDoc(docRef); if (!docSnap.exists()) { container.innerHTML = "

Produto não existe

"; return; } const data = docSnap.data(); container.innerHTML = `

${data.titulo}

R$ ${data.preco}

Descrição: ${data.descricao}

📍 ${data.cidade}

🏷 ${data.categoria}

📊 Score: ${data.score}%

`; } catch (error) { console.error(error); container.innerHTML = "

Erro ao carregar produto

"; } } carregarProduto();
